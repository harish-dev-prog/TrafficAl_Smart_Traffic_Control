🧠 AI Algorithms Used in Traffic Control
Greedy Algorithm — Signal Phase Optimization

At each cycle, the AI picks the lane with maximum queued vehicles to get green signal — a greedy local optimal choice.

// Greedy: Give green to most congested lane
function aiOptimize(lanes) {
  let maxQueue = 0, bestLane = null;
  for (let lane of lanes) {
    if (lane.queueLength > maxQueue) {
      maxQueue = lane.queueLength;
      bestLane = lane;
    }
  }
  return bestLane; // give green signal
}
Priority Queue — Emergency Vehicle Preemption

Emergency vehicles use a Max-Priority Queue. Ambulance/Fire (Critical) always gets cleared first, overriding normal signal phases.

// Priority Queue — higher priority = served first
class PriorityQueue {
  constructor() { this.heap = []; }
  insert(unit) {
    this.heap.push(unit);
    this.heap.sort((a,b) => b.priority - a.priority);
  }
  extractMax() { return this.heap.shift(); }
}
// Critical=3, High=2, Medium=1
Dijkstra's Algorithm — Shortest Emergency Route

When an emergency unit is dispatched, Dijkstra finds the shortest path through intersections, clearing signals along that route.

// Dijkstra's shortest path between intersections
function dijkstra(graph, source) {
  let dist = {}, visited = new Set();
  for (let node in graph) dist[node] = Infinity;
  dist[source] = 0;
  while (visited.size < Object.keys(graph).length) {
    let u = minDist(dist, visited);
    visited.add(u);
    for (let v in graph[u]) {
      if (dist[u] + graph[u][v] < dist[v])
        dist[v] = dist[u] + graph[u][v];
    }
  }
  return dist;
}
BFS — Intersection Reachability Check

Breadth-First Search checks which intersections are reachable from any point — used to detect road closures and alternate routes.

// BFS traversal on road network graph
function bfsReachable(graph, start) {
  let visited = [], queue = [start];
  while (queue.length) {
    let node = queue.shift();
    if (!visited.includes(node)) {
      visited.push(node);
      queue.push(...graph[node]);
    }
  }
  return visited;
}

🗂️ Data Structures in Traffic Control
Queue — Vehicle Lane Management

Each traffic lane is modeled as a FIFO Queue. Vehicles enqueue when they arrive and dequeue when the signal turns green — classic queue behavior.

// FIFO Queue — Lane at an intersection
class LaneQueue {
  constructor(dir) { this.lane=[]; this.dir=dir; }
  arrive(vehicle) { this.lane.push(vehicle); }     // enqueue
  depart()        { return this.lane.shift(); }    // dequeue (green light)
  size()          { return this.lane.length; }
}
// North, South, East, West — 4 queues per intersection
Stack — Signal History / Undo

Signal state changes are pushed onto a stack, enabling rollback to previous configuration during system faults.

// Stack — signal phase rollback
class SignalHistory {
  constructor() { this.stack = []; }
  save(state) { this.stack.push(state); }
  undo()      { return this.stack.pop(); }
  peek()      { return this.stack[this.stack.length-1]; }
}
Hash Map — Intersection Lookup

O(1) average-case lookup of any intersection's current signal state using intersection ID as key.

// HashMap — instant intersection state lookup
const intersectionMap = new Map();
intersectionMap.set("I001", { state:"green", timer:30 });
intersectionMap.set("I002", { state:"red",   timer:15 });

// O(1) lookup — no loop needed
let signal = intersectionMap.get("I001");
Linked List — Event Log Chain

System events (signal changes, emergency clears) are stored as a singly linked list — O(1) prepend for new events.

// Singly Linked List — event log
class EventNode {
  constructor(event) { this.event=event; this.next=null; }
}
class EventLog {
  constructor() { this.head=null; this.size=0; }
  prepend(event) {
    let node = new EventNode(event);
    node.next = this.head;
    this.head = node;
    this.size++;
  }
}

Intersections = Vertices. Roads = Edges. Travel time / distance = Edge weights. Directed graph for one-way roads.

Adjacency Matrix — Signal Dependency

A matrix showing which lanes conflict — conflicting lanes cannot both be green simultaneously.

// Conflict matrix for a 4-lane intersection
//         N  E  S  W
const conflict = [

// 0 = safe together, 1 = conflict (cannot both be green)
DFS — Detecting Deadlock / Cycles

Depth-First Search detects cycles in the signal dependency graph — if a cycle exists, deadlock is possible and AI breaks it.

// DFS cycle detection on signal graph
function hasCycle(graph) {
  let visited={}, recStack={};
  function dfs(v) {
    visited[v]=true; recStack[v]=true;
    for (let w of (graph[v]||[])) {
      if (!visited[w] && dfs(w)) return true;
      if (recStack[w]) return true; // cycle!
    }
    recStack[v]=false; return false;
  }
  for (let v in graph) { if(!visited[v]&&dfs(v)) return true; }
  return false;
}
